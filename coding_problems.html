Problem 1 

public static String landPerimeter(String[] arr) {
    Integer round = 0; Integer result = 0;
    HashMap<Integer, HashSet<Integer>> levelAndIndex = new HashMap<>();

    for(int i = 0; i < arr.length; i++){
      HashSet<Integer> indexLevels = new HashSet<>();
      char[] individ = arr[i].toCharArray();

      for(int k = 0; k < individ.length; k++){
        if(individ[k] == 'X'){
          indexLevels.add(k);
        }
      }
      round++;
      levelAndIndex.put(round, indexLevels);
    }

    for(Integer i : levelAndIndex.keySet()){
      HashSet<Integer> copy = new HashSet<>(levelAndIndex.get(i));

      for(Integer u : copy){
        if(copy.contains(u)){
          result += 4;
          if(u > 0 && copy.contains(u+1)){
            result -= 2;
          }else if(u > 0 && copy.contains(u-1)){
            result -=2;
          }
        }
      }
    }
    return "Total land perimeter is: "+result;
  }

The objective of this problem was to calculate the land perimeter of given X and O array. Each tile had a 1x1 measurement.

public static String order(String words) {
    String[] wrd = words.split(" ");
    HashMap <String, Integer> x = new HashMap<String, Integer>();
    LinkedHashMap<String, Integer> sortedMap = new LinkedHashMap<>();
    ArrayList<Integer> list = new ArrayList<>();
    String res = "";
    int k = 0; 
    for(String c : wrd){
      for(int i = 0; i<c.length(); ++i){
        if(Pattern.matches("[123456789]", String.valueOf(c.charAt(i)))){
          x.put(c, Integer.parseInt(String.valueOf(c.charAt(i)))));
        }
      }
    }
    
    for (Map.Entry<String, Integer> entry : x.entrySet()) {
      list.add(entry.getValue());
    }
    Collections.sort(list); 
    for (int num : list) {
      k++;
        for (Entry<String, Integer> entry : x.entrySet()) {
            if (entry.getValue().equals(num)) {
                res += k == list.size()? entry.getKey() : entry.getKey() + " ";
                sortedMap.put(entry.getKey(), num);
            }
        }
    }

    return res;
  }

The objective of this program is to sort a given string with a randomized number embedded into it.
